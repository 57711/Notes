# 闭包陷阱

## useState()

```js
function Index() {
  const [count, setCount] = useState(0);

  function add(){
   setCount( count + 1 );
   setCount((count) => count + 1);
   setCount( count + 1 );
   console.log(count);  // 0
  }
  return (
   <div>
  <span>{count}</span>
  <button @click=()=>{ add() }> + </button>
  </div>
  );
}
```

## useEffect()

```js
function Index() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    setInterval(() => {
      console.log(`Count: ${count}`);
    }, 1000);
  }, []);

  return (
    <div>
      <span>{count}</span>
      <button onClick={() => setCount(count + 1)}> + </button>
    </div>
  );
}
```

用`useRef()` 缓存每一次 count 更新

```js
function Index() {
  const [count, setCount] = useState(0);
  const countRef = useRef();
  countRef.current = count;

  useEffect(() => {
    setInterval(() => {
      console.log(`Count: ${countRef.current}`);
    }, 1000);
  }, []);

  return (
    <div>
      <span>{count}</span>
      <button onClick={() => setCount(count + 1)}> + </button>
    </div>
  );
}
```
